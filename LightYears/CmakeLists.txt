cmake_minimum_required(VERSION 4.1.2)

project(LightYears
	VERSION 1.0.0
	LANGUAGES C CXX
)

set(CMAKE_CXX_STANDARD			14)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS		OFF)

# Fetch SFML #
include(FetchContent)
set(SFML_LIB_NAME SFML)
FetchContent_Declare(
	${SFML_LIB_NAME}
	GIT_REPOSITORY https://github.com/SFML/SFML
	GIT_TAG 3.0.1
	GIT_SHALLOW FALSE
)
FetchContent_MakeAvailable(${SFML_LIB_NAME})

# Creates a variable (NAME value). To use it: ${NAME}
set(LIGHT_YEARS_GAME_TARGET_NAME LightYearsGame)

# The following subdir holds the source files used for this project.
# If more projects are needed/created in the future, then simply create another subdir and add another CmakeLists that holds what's needed for that project!
add_subdirectory(LightYears)
